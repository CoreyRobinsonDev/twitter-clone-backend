users 
  id INTEGER
  username TEXT
  password TEXT
  profile_photo TEXT
  banner_photo TEXT
  bio TEXT
  num_tweets INTEGER
  num_followers INTEGER
  num_following INTEGER
  date_acc_created NUMERIC (mm/dd/yyyy)
  
  PRIMARY KEY(id)


posts
  id INTEGER
  poster_id INTEGER
  text TEXT
  media TEXT
  date_post_created NUMERIC (unix timestamp)
  num_comments INTEGER
  num_upvotes INTEGER
  num_downvotes INTEGER
  num_reposts INTEGER

  PRIMARY KEY(id)
  FOREIGN KEY(poster_id) REFERENCES users(id)


comment_section
  id INTEGER
  poster_id INTEGER
  post_id INTEGER
  text TEXT
  media TEXT
  num_upvotes INTEGER
  num_downvotes INTEGER
  num_reposts INTEGER

  PRIMARY KEY(id)
  FOREIGN KEY(poster_id) REFERENCES users(id)
  FOREIGN KEY(post_id) REFERENCES posts(id)


bookmarks
  user_id INTEGER
  post_id INTEGER

  FOREIGN KEY(user_id) REFERENCES users(id)
  FOREIGN KEY(post_id) REFERENCES posts(id)


likes
  user_id INTEGER
  post_id INTEGER

  FOREIGN KEY(user_id) REFERENCES users(id)
  FOREIGN KEY(post_id) REFERENCES posts(id)


notifications


followers
  user_id INTEGER
  follower_id INTEGER

  FOREIGN KEY(user_id) REFERENCES users(id)
  FOREIGN KEY(follower_id) REFERENCES users(id)


messages
  conversation_id INTEGER
  content TEXT
  time NUMERIC

  FOREIGN KEY(conversation_id) REFERENCES conversations(id)


conversations
  id INTEGER
  sender_id INTEGER
  receiver_id INTEGER

  PRIMARY KEY(id)
  FOREIGN KEY(sender_id) REFERENCES users(id)
  FOREIGN KEY(receiver_id) REFERENCES users(id)

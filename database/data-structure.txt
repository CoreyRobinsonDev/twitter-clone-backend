users 
  id INTEGER
  username TEXT NOT NULL
  password TEXT NOT NULL
  profile_photo TEXT NOT NULL
  banner_photo TEXT NOT NULL
  bio TEXT
  num_tweets INTEGER NOT NULL
  num_followers INTEGER NOT NULL
  num_following INTEGER NOT NULL
  date_acc_created NUMERIC NOT NULL (mm/dd/yyyy)
  
  PRIMARY KEY(id)


posts
  id INTEGER
  poster_id INTEGER NOT NULL
  text TEXT
  media TEXT
  media_content_type TEXT
  date_post_created NUMERIC NOT NULL (unix timestamp)
  num_comments INTEGER NOT NULL
  num_upvotes INTEGER NOT NULL
  num_downvotes INTEGER NOT NULL
  num_reposts INTEGER NOT NULL

  PRIMARY KEY(id)
  FOREIGN KEY(poster_id) REFERENCES users(id)


comment_section
  id INTEGER
  poster_id INTEGER
  post_id INTEGER
  text TEXT
  media TEXT
  num_upvotes INTEGER
  num_downvotes INTEGER
  num_reposts INTEGER

  PRIMARY KEY(id)
  FOREIGN KEY(poster_id) REFERENCES users(id)
  FOREIGN KEY(post_id) REFERENCES posts(id)


bookmarks
  user_id INTEGER NOT NULL
  post_id INTEGER
  comment_id INTEGER

  FOREIGN KEY(user_id) REFERENCES users(id)
  FOREIGN KEY(post_id) REFERENCES posts(id)
  FOREIGN KEY(comment_id) REFERENCES comment_section(id)

upvotes
  user_id INTEGER NOT NULL
  post_id INTEGER
  comment_id INTEGER

  FOREIGN KEY(user_id) REFERENCES users(id)
  FOREIGN KEY(post_id) REFERENCES posts(id)
  FOREIGN KEY(comment_id) REFERENCES comment_section(id)

downvotes
  user_id INTEGER NOT NULL
  post_id INTEGER
  comment_id INTEGER

  FOREIGN KEY(user_id) REFERENCES users(id)
  FOREIGN KEY(post_id) REFERENCES posts(id)
  FOREIGN KEY(comment_id) REFERENCES comment_section(id)

reposts
  user_id INTEGER NOT NULL
  post_id INTEGER
  comment_id INTEGER
  date_reposted NUMERIC NOT NULL (unix timestamp)

  FOREIGN KEY(user_id) REFERENCES users(id)
  FOREIGN KEY(post_id) REFERENCES posts(id)
  FOREIGN KEY(comment_id) REFERENCES comment_section(id)

notifications


followers
  user_id INTEGER
  follower_id INTEGER

  FOREIGN KEY(user_id) REFERENCES users(id)
  FOREIGN KEY(follower_id) REFERENCES users(id)


messages
  conversation_id INTEGER
  content TEXT
  time NUMERIC

  FOREIGN KEY(conversation_id) REFERENCES conversations(id)


conversations
  id INTEGER
  sender_id INTEGER
  receiver_id INTEGER

  PRIMARY KEY(id)
  FOREIGN KEY(sender_id) REFERENCES users(id)
  FOREIGN KEY(receiver_id) REFERENCES users(id)
